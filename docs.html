Getting Started

Scene

Create a Scene

let visual = new Visual ( mouseRadius, canvasName, z-index, backgroundColor, messages);
new Visual ( Integer, String, Integer, String, Boolean);



Create Lights

visual.softLight( radius, Color, intensity, position, attachMouse, PosX, PosY);
variable.softLight( Integer, String, Integer, String, Boolean, Integer, Integer);

visual.spotLight( radius, Color, intensity, position, attachMouse, PosX, PosY);
variable.spotLight( Integer, String, Integer, String, Boolean, Integer, Integer);

visual.flashLight( radius, Color, intensity, position, attachMouse, PosX, PosY);
variable.flashLight( Integer, String, Integer, String, Boolean, Integer, Integer);



Move Lights

visual.startLight( id, {move: {x: PosX, y: PosY}}, duration, yoyo); visual.startLight( String, {move: {x: Integer, y: Integer}}, Integer, Boolean);

let particle = new Particle({ amount: Amount, color: Color, radius: radius, pos: { x: PosX, y: PosY }, collision: { collisionMouse: CollisionMouse, collisionBorder: CollisionBorder, collisionOther: CollisionOther }, connect: { connectColor: ConnectColor, connectNear: ConnectNear, connectMouse: ConnectMouse, connectRadius: ConnectRadius }, negativeX: NegativeX, negativeY: NegativeY });
